  \begin{enumerate}
   \item $M(x;y)$ dans le repère $(O;\vec{i},\vec{j}) \Leftrightarrow \vec{OM}=x \vec{i}+y \vec{j}$. 
   \item $\vec{u}(x;y)$ et $\vec{u'}(x';y')$ sont colinéaires si et seulement si $xy'-x'y=0$.
   \item Si $\vec{u}$ dirige $d$ et $\vec{u'}$ dirige $d'$:
   $d // d' \Leftrightarrow$ $\vec{u}$ et $\vec{u'}$ sont colinéaires. 
   \item $d:ax+by+c=0 \Leftrightarrow \vec{u}{(-b;a)}$ dirige $d$.
   \item $d:y=mx+p \Leftrightarrow \vec{u}(1;m)$ dirige $d$.
   \item $d:x=k \Leftrightarrow \vec{u}{(0;1)}$ dirige $d$.
  \end{enumerate}
